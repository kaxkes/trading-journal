function doPost(e) {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  let sh = ss.getSheetByName('Trades');
  if (!sh) {
    sh = ss.insertSheet('Trades');
    sh.appendRow(['Id','Date','Asset','Direction','Session',
      'Result','Rules','Rules_count',
      'Emo_before','Emo_during','Timestamp']);
  }
  const d = JSON.parse(e.postData.contents);
  sh.appendRow([d.id, d.date, d.asset, d.direction, d.session,
    d.result, d.rules, d.rules_count,
    d.emo_before, d.emo_during,
    new Date().toLocaleString('th-TH',
      {timeZone:'Asia/Bangkok'})]);
  return ContentService
    .createTextOutput(JSON.stringify({ok:true}))
    .setMimeType(ContentService.MimeType.JSON);
}

function doGet(e) {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const sh = ss.getSheetByName('Trades');
  if (!sh) {
    return ContentService
      .createTextOutput(JSON.stringify({trades:[]}))
      .setMimeType(ContentService.MimeType.JSON);
  }
  const rows = sh.getDataRange().getValues();
  const header = rows[0];
  const dateColIdx = header.indexOf('Date');
  const trades = rows.slice(1).map(r => {
    const o = {};
    header.forEach((h,i) => o[h] = r[i]);
    // กันเหนียว: ถ้า Sheets auto-convert ช่อง Date เป็น Date object ให้ format กลับเป็น YYYY-MM-DD (ใช้เวลาไทยตรงๆ ไม่ผ่าน UTC เพื่อกัน off-by-one)
    let dateStr;
    if (dateColIdx > -1 && r[dateColIdx] instanceof Date) {
      dateStr = Utilities.formatDate(r[dateColIdx], 'Asia/Bangkok', 'yyyy-MM-dd');
    } else {
      dateStr = String(o.Date || '');
    }
    return {
      id: o.Id,
      date: dateStr,
      asset: o.Asset,
      direction: o.Direction,
      session: o.Session,
      result: o.Result,
      rules: o.Rules ? String(o.Rules).split(',').map(s=>s.trim()).filter(Boolean) : [],
      rules_count: o.Rules_count,
      emo_before: o.Emo_before,
      emo_during: o.Emo_during,
      synced: true
    };
  });
  return ContentService
    .createTextOutput(JSON.stringify({trades}))
    .setMimeType(ContentService.MimeType.JSON);
}
